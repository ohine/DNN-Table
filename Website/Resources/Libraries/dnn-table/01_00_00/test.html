dnnTable Example 1

Initializing the dnnTable with an array of objects. Also featuring the render ability with columns 

<table id="example1"></table>
<script type="text/javascript">
(function ($) {
  $('#example1').dnnTable({
    'data': [
      { 'id': 1, 'firstName': 'Chris', 'lastName': 'Marfia', 'dob': '05/22/1991' },
      { 'id': 2, 'firstName': 'Joe', 'lastName': 'Zacher', 'dob': '06/31/1990' },
      { 'id': 3, 'firstName': 'Kristen', 'lastName': 'Ritchie', 'dob': '03/06/1998' },
      { 'id': 4, 'firstName': 'Oliver', 'lastName': 'Hine', 'dob': '02/12/1985' }
    ],
    'totalRecordCount': 4,
    'primaryKeyProp': 'id',
    'columns': [
      { 'title': 'Student ID', 'propName': 'id' },
      { 'title': 'Full Name', 'render': function (dataRow) { 
   	    return dataRow['firstName'] + ' ' + dataRow['lastName']; 
      }},
      { 'title': 'Date of Birth', 'propName': 'dob' }
    ]
  });
}(jQuery));
</script>
